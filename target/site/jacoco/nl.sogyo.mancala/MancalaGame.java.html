package nl.sogyo.mancala;

public class MancalaGame implements Mancala {
    private static Bowl bowl = new Bowl();
    public String getPlayerName(int playerIndex) throws IllegalStateException {
        if(playerIndex==1){
            return bowl.getOwner().getName();
        } else if(playerIndex==2) {
            return bowl.getOwner().getOpponent().getName();
        } else {
            throw new IllegalStateException();
        }
    }

    public void setPlayerName(String name, int playerIndex) throws IllegalStateException {
        if(playerIndex==1){
            bowl.getOwner().setName(name);
        } else if(playerIndex==2) {
            bowl.getOwner().getOpponent().setName(name);
        } else {
            throw new IllegalStateException();
        }
    }

    public boolean isToMovePlayer(int playerIndex) throws IllegalStateException {
        if(playerIndex==1){
            return bowl.getOwner().getIsMyTurn();
        } else if(playerIndex==2) {
            return bowl.getOwner().getOpponent().getIsMyTurn();
        } else {
            throw new IllegalStateException();
        }
    }

    public int[] playRecess(int index) {
        AbstractBowl bowlToPlay = bowl.getNeighbour(index-1);
        if (bowlToPlay.getOwner().getIsMyTurn() && (bowlToPlay instanceof Bowl) && bowlToPlay.getBeads()!=0){
            ((Bowl) bowlToPlay).play();
        }
        return showBoard();
    }

    public int getStonesForPit(int index) {
        return bowl.getNeighbour(index-1).getBeads();
    }

    public int[] exportGameState() {
        return showBoard();
    }

    public boolean isEndOfGame() {
        return bowl.isSideCurrentPlayerEmpty();
    }

    public String getWinnersName() {
        if(!isEndOfGame()){return null;}
        int beadsPlayer1 = bowl.getNeighbour(6).getBeads();
        int beadsPlayer2 = bowl.getNeighbour(13).getBeads();
        if(beadsPlayer1 > beadsPlayer2){
            return bowl.getNeighbour(6).getOwner().getName();
        } else if(beadsPlayer2 > beadsPlayer1) {
            return bowl.getNeighbour(13).getOwner().getName();
        } else {return "It's a tie!";}
    }

    private int[] showBoard(){
        int[] boardStatus = new int[15];
        for(int i=0;i<14;i++){
            boardStatus[i] = bowl.getNeighbour(i).getBeads();
        }
        boardStatus[14] = (bowl.getOwner().getIsMyTurn() ? 1 : 2);
        return boardStatus;
    }
}
